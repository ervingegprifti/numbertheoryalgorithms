\section*{Mod Factors}	
Any number $n$ can be written as $(ax+c)(ay+b)=n$. Distributing we have	$a(axy+bx+cy) + bc=n$. Given any $a$ we can find $b,c$ such as $n \equiv bc \pmod{a}$. \\




\begin{algorithm}[H]
	\caption{Mod Factors}
	\DontPrintSemicolon
	\SetAlgoLined
	\KwIn{$n,a \in \mathbb{N}$}
	\KwOut{$b,c$ such as $n \equiv bc \pmod{a}$}
	
	\BlankLine
	
	Let $n \pmod{a} = r$ \\
	For every $b,c$ combination where $b,c \in \left\{0, \dots, a-1\right\}$ \\
	\lIf{$bc \pmod{a} = r$} { Display and count. }
	
	\Return{$b,c$ such as $n \equiv bc \pmod{a}$}
\end{algorithm}
